
**方法一：PropTypes（適合 JavaScript 初學者）**

    import PropTypes from 'prop-types';
    
    function Header({ title }) {
      return <h1>{title}</h1>;
    }
    
    Header.propTypes = {
      title: PropTypes.string.isRequired,
    };

**方法二：TypeScript（適合更嚴謹的開發）**

    type HeaderProps = {
      title: string;
    };
    
    function Header({ title }: HeaderProps) {
      return <h1>{title}</h1>;
    }
